    p.product_name, 
    SUM(o.total_price) AS revenue
FROM products p
JOIN order_items o ON p.id = o.product_id
WHERE o.created_at >= '2023-10-01'
GROUP BY 1
ORDER BY 2 DESC
LIMIT 10;
